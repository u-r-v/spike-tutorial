LEGO Learning System (LLS) Python Tutorial

Heads up: This guide uses the current LLS Python APIs (hub, motor, motor_pair, runloop, etc.). Older spike imports have been removed.

Program structure

+

Wrap your script in async def main() and start it with runloop.run(main()).

+ 
import runloop
+
+async def main():
+    # your code here
+    await runloop.sleep_ms(500)
+
+runloop.run(main())
+ +

Example 1 — Single motor (“Hello, motors”)

+ 
from hub import port
+import runloop, motor
+
+async def main():
+    # One full rotation at 500 deg/s
+    await motor.run_for_degrees(port.A, 360, 500)
+    await runloop.sleep_ms(500)
+    # Park back to 0 degrees at 400 deg/s
+    await motor.run_to_absolute_position(port.A, 0, 400)
+    motor.stop(port.A)
+
+runloop.run(main())
+ +

Example 2 — Driving base (paired motors)

+ 
from hub import port
+import runloop, motor_pair
+
+# Pair once and reuse the pair ID
+motor_pair.pair(motor_pair.PAIR_1, port.A, port.E)
+
+async def main():
+    # Straight ~10 cm (≈ 206 deg for SPIKE wheels)
+    await motor_pair.move_for_degrees(motor_pair.PAIR_1, 206, 0, velocity=300)
+    await runloop.sleep_ms(300)
+    # Right turn using steering
+    await motor_pair.move_for_degrees(motor_pair.PAIR_1, 182, 100, velocity=300)
+    motor_pair.stop(motor_pair.PAIR_1)
+
+runloop.run(main())
+ +

Example 3 — Color sensor: stop on line, simple “bounce” follow

+ 
from hub import port, button
+import runloop, motor_pair, color_sensor, color
+
+motor_pair.pair(motor_pair.PAIR_1, port.A, port.E)
+
+async def main():
+    # Stop on black line when LEFT is held
+    if button.pressed(button.LEFT):
+        motor_pair.move(motor_pair.PAIR_1, 0, velocity=300)
+        await runloop.until(lambda: color_sensor.color(port.B) is color.BLACK)
+        motor_pair.stop(motor_pair.PAIR_1)
+
+    # Simple "bounce" line follow when RIGHT is held
+    if button.pressed(button.RIGHT):
+        while True:
+            motor_pair.move_tank(motor_pair.PAIR_1, 0, 300)
+            await runloop.until(lambda: color_sensor.color(port.B) is color.BLACK)
+            motor_pair.move_tank(motor_pair.PAIR_1, 300, 0)
+            await runloop.until(lambda: color_sensor.color(port.B) is color.WHITE)
+
+runloop.run(main())
+ +

Example 4 — Distance sensor: approach and stop

+ 
from hub import port
+import runloop, motor_pair, distance_sensor
+
+motor_pair.pair(motor_pair.PAIR_1, port.A, port.E)
+
+async def main():
+    motor_pair.move(motor_pair.PAIR_1, 0, velocity=300)
+    # Stop when within 100 mm of an object
+    await runloop.until(lambda: distance_sensor.distance(port.F) < 100)
+    motor_pair.stop(motor_pair.PAIR_1)
+
+runloop.run(main())
+ +

Example 5 — Hub lights & sounds

+ 
from hub import light_matrix, sound
+import runloop
+
+async def main():
+    light_matrix.show_image(light_matrix.IMAGE_HEART)
+    await sound.beep(523, 200)   # C5, 200 ms
+    await runloop.sleep_ms(500)
+    light_matrix.show_image(light_matrix.IMAGE_NO)
+
+runloop.run(main())

Notes

+
    +
  • Async first: use await runloop.sleep_ms(...) instead of blocking waits.
    • +
  • Distances: distance_sensor.distance() returns millimeters.
    • +
  • Colors: compare with color.<NAME> enums (e.g., color.BLACK).
    • +
  • Motor pairs: pair once with motor_pair.pair(...) and reuse the pair id.
