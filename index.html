Motors

from hub import port
import runloop, motor

async def main():
    # 1 rotation at 500 deg/s
    await motor.run_for_degrees(port.A, 360, 500)
    await runloop.sleep_ms(1000)
    motor.stop(port.A)

runloop.run(main())

Motor Pair

from hub import port
import runloop, motor_pair

# Pair your drive motors once and reuse the pair ID
motor_pair.pair(motor_pair.PAIR_1, port.A, port.E)

async def main():
    # Start driving straight (steering 0) at 500 deg/s
    motor_pair.move(motor_pair.PAIR_1, 0, velocity=500)
    await runloop.sleep_ms(1000)
    motor_pair.stop(motor_pair.PAIR_1)

runloop.run(main())

Color Sensor

from hub import port
import color_sensor, color

# Read the detected color on port B
detected = color_sensor.color(port.B)
print(detected is color.BLACK)

Distance Sensor

from hub import port
import distance_sensor

# Distance is returned in millimeters
mm = distance_sensor.distance(port.F)
print(mm)
